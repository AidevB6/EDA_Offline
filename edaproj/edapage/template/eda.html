f, ax = plt.subplots(1,2,figsize=(20,8))

labels = df['threat_level'].value_counts().index
frequency = df['threat_level'].value_counts()
pie = ax[0].pie(frequency, startangle=90, counterclock=False, 
                wedgeprops=dict(width=0.5, edgecolor='w'))
 
total = np.sum(frequency) 
 
sum_pct = 0 
for i,l in enumerate(labels):
    ang1, ang2 = pie[0][i].theta1, pie[0][i].theta2 
    r = pie[0][i].r 
    
    x = ((r+0.5)/2)*np.cos(np.pi/180*((ang1+ang2)/2)) 
    y = ((r+0.5)/2)*np.sin(np.pi/180*((ang1+ang2)/2)) 
    
    if i < len(labels) - 1:
        sum_pct += float(f'{frequency[i]/total*100:.2f}') 
        ax[0].text(x,y,f'{frequency[i]/total*100:.2f}%',ha='center',va='center', fontsize=12) 
    else:
        ax[0].text(x,y,f'{100-sum_pct:.2f}%',ha='center',va='center', fontsize=12)
        
ax[0].set_title('Threat Level', fontsize=20)
ax[0].set_ylabel('')
ax[0].legend(labels, fontsize=12)

sns.countplot('threat_level', data=df, ax=ax[1])
ax[1].set_title('Threat Level', fontsize=20)
ax[1].set_xlabel('Threat Level', fontsize=18)
ax[1].set_ylabel('Count', fontsize=18)
ax[1].tick_params(axis='both', labelsize=15)

나이와 무장 종류의 관계¶

  • 총 (Gun) : 총기 허용 국가인 만큼 거의 전 연령대에서 범죄자가 총기를 사용한 경우가 존재
  • 비무장 (Unarmed) : Teenage 영역에서도 비무장 상태에서 사망한 범죄자가 발생한 경우가 존재
  • 장난감 (Toy) : 주변에서 쉽게 볼수 있는 물건으로 대부분의 연령대에 존재
  • 단도 (Knife) : 주변에서 쉽게 볼수 있는 물건으로 대부분의 연령대에 존재
  • 장검 (Sword), 도끼(Axe), 마체테(Machete) : 크기가 크고 무겁기에 대부분 20대 ~ 60대에 존재
  • 테이저건 (Taser) : 가장 좁은 나이 범위에서 발생
